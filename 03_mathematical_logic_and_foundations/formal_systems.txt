                        1               := one
                        +               := plus
                        ×               := times

                        fixed for first-order logic
                        =               := equality

                        ∨               := or
                        ¬               := not
                        ∃               := exists
                        ∀               := forall
                        x,y,z ...       := variable symbols
                        (,)             := parenthesis

                def grammar: how you can put the symbols together.
