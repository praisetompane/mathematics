quantifiers(Kunen, 2007:4):
    ∀ := for all
    ∃ := there exists
    ∃! := there exists a unique one

    properties:
        - domain of discourse or universe required
            if unspecified, the Real Numbers are implied(Kunen, 2007).
            example:
                ∀x[x² > 4 ⟹ [x > 2 ∨ x < -2]]
                is actually:
                ∀x ∈ ℝ[x² > 4 ⟹ [x > 2 ∨ x < -2]]
        - presumes (Kunen, 2007):
            - functions/operators such as ∧ ∨ > <
                have precise meaning
                    in the domain of discourse
            - objects used such as 2, 4, +-2
                exist
                    in the domain of discourse
    example usage:
        - ∀ := for all:
            for all Real numbers,
                the sum of two Real numbers
                    is a Real number.
                ∀a,b ∈ ℝ, a+b ∈ ℝ
        
        - ∃ := there exists:
            for all Rational numbers
                there exists an inverse.
                ∀a ∈ ℚ, ∃a⁻¹ such that a * a⁻¹ = 1
            
        - ∃! := there exists a unique one         
            every positive Natural numbers
                has a positive square root.
                ∀a ∈ ℕ a > 0, ∃!b s.t. b² = a ∧ b > 0
    NB: often omitted 
        example:
            commutative law: 
                x ⋅ y = y ⋅ x
                is equivalent to
                ∀x,y ∈ ℝ, x ⋅ y = y ⋅ x
 
number sets:
    ℕ := set of natural numbers
    ℤ⁺:= set of positive integers
    ℤ := set of integers
    ℝ := Real Numbers
    ℚ := Rational Numbers

def sentence:   a statement with no free variables.
                    in a given universe
                        it MUST be True or False (Kunen, 2007:4).
                
                NB: The statement is True or False for the whole universe.
            
                Example:
                    ∀x[x > 0 ⟹ ∃!y[y² = x ∧ y > 0]] (Kunen, 2007:4):
                        Value       Universe
                          True         ℕ
                          False        ℝ

def formula:    a statement that makes an assertion about a variable.
                    in a given universe 
                        it MAY be True for some values of the variable
                        or MAY be False for some values of the variable(Kunen, 2007:4)
                        
∴        := therefore
ⁿ√x      := nᵗʰ root of x

propositional connectives(Kunen, 2007:4):
    ∧       := and
    ∨       := or
    ¬       : not | negation
    ⟹       : implies
    ⟺       : if and only if

    truth table(Kunen, 2007:4): 
        def T: True
        def F: False
        
        φ   ψ   φ ∨ ψ    φ ∧ ψ      φ ⟹ ψ   φ ⟺ ψ
        T   T   T        T          T        T
        T   F   T        F          F        F
        F   T   T        F          T        F     
        F   F   F        F          T        T

        φ   ¬φ
        T   F
        F   T

        NB: φ ⟹ ψ is always equivalent to ¬φ ∨ ψ
