quantifiers(Kunen, 2007):
    ∀ := for all
    ∃ := there exists

    properties:
        - domain of discourse or universe required
            if unspecified, the Real Numbers are implied(Kunen, 2007).
            example:
                ∀x[x² > 4 ⟹ [x > 2 ∨ x < -2]]
                is actually:
                ∀x∈ℝ[x² > 4 ⟹ [x > 2 ∨ x < -2]]
        - presumes (Kunen, 2007):
            - functions/operators such as ∧ ∨ > <
                have precise meaning
                    in the domain of discourse
            - objects used such as 2, 4, +-2
                exist
                    in the domain of discourse

number sets:
    ℕ := set of natural numbers
    ℤ⁺:= set of positive integers
    ℤ := set of integers
    ℝ := Real Numbers
    ℚ := Rational Numbers

∴        := therefore
ⁿ√x      := nᵗʰ root of x

constructors:
    {}       := Set
    ...      := infinite, with same pattern
    :=       := assignment

propositional connectives:
    ∧       := and
    ∨       := or
    ¬       : not | negation
    ⟹       : implies
    ⟺       : if and only if

logical operators:
    > greater than
    < less than
    = equal
    ≥ greater than or equal
    ≤ less than or equal

set operators:
    \   := minus
    ∪   := addition
    ∈   := in (is element of)
