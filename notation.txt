quantifier:
    ∀ := for all := quantifier
    ∃ := there exist

number sets:
    ℕ := set of narutal numbers
    ℤ⁺:= set of positive integers
    ℤ := set of integers
    ℝ := Real Numbers
    ℚ := Rational Numbers

∴        := therefore

constructors:
    {}       := Set
    ...      := infinite, with same pattern
    :=       := assignment

logical operators:
    ∧        := and
    ∨        := or
    ⟹       := implies
    ⟺       := if and only if

set operators:
    \   := minus
    ∪   := addition
    ∈   := in (is element of)
